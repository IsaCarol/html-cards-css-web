Card Simples — _HTML.CARD

Card básico com título, valor principal (callout) e tooltip opcional ativado pelo botão "i". A barra colorida lateral muda automaticamente conforme a posição do card.

-- Medida: [Cards - KPIs Resumo]
VAR _CSS   = _CSS.CARD.STYLE(0)

VAR _Cards = _HTML.CARD(
    {
        -- ( Título,          Valor,              Formato,       Tooltip )
        ("Receita Total",  [Receita Total],    "R$ #,0",      "Receita bruta acumulada no período."),
        ("Custo Total",    [Custo Total],      "R$ #,0",      BLANK()),
        ("Margem Bruta",   [Margem Bruta %],   "0.0%",        BLANK()),
        ("Nº de Pedidos",  [Total Pedidos],    "#,0",         BLANK()),
        ("Ticket Médio",   [Ticket Médio],     "R$ #,0.00",   BLANK())
    }
)

RETURN _HTML.CSS.CARD.CONSTRUCTOR(_CSS, _Cards)

Card com Sub-Métricas — _HTML.CARD.MULTI_LABEL

Exibe rótulos adicionais abaixo do valor principal. Ideal para comparativos como Ano Anterior, Meta, Mês Anterior etc.

VAR _CSS = _CSS.CARD.STYLE(0)

VAR _Callout = {
    ("Receita",  [Receita Total],  "R$ #,0",  BLANK()),
    ("Custo",    [Custo Total],    "R$ #,0",  BLANK()),
    ("Margem",   [Margem %],       "0.0%",    BLANK())
}

-- Value1 deve ser idêntico ao título do card pai (chave de join)
VAR _Labels = {
    ("Receita",  "vs. Ano Ant.",  [Receita AA],  "R$ #,0",  "Período equivalente"),
    ("Receita",  "Meta",          [Meta Receita],"R$ #,0",  BLANK()),
    ("Custo",    "vs. Ano Ant.",  [Custo AA],    "R$ #,0",  BLANK()),
    ("Margem",   "vs. Ano Ant.",  [Margem AA],   "0.0%",    BLANK())
}

RETURN _HTML.CSS.CARD.CONSTRUCTOR(
    _CSS,
    _HTML.CARD.MULTI_LABEL(_Callout, _Labels)
)

Card com Badge de Variação — _HTML.CARD.MULTI_LABEL.DETAIL

Adiciona um badge colorido com seta automática (▲/▼) em cada sub-métrica. A cor reage ao sinal do ValorParaCor — permitindo inverter a polaridade para métricas como Custo ou Devoluções.

VAR _CSS     = _CSS.CARD.STYLE(0)
VAR _Callout = { ("Receita", [Receita], "R$ #,0", BLANK()), ("Custo", [Custo], "R$ #,0", BLANK()) }
VAR _Labels  = { ("Receita", "vs. Ano Ant.", [Rec AA], "R$ #,0", BLANK()), ("Custo", "vs. Ano Ant.", [Custo AA], "R$ #,0", BLANK()) }

VAR _VarRec   = DIVIDE([Receita] - [Rec AA],   [Rec AA])
VAR _VarCusto = DIVIDE([Custo]   - [Custo AA], [Custo AA])

VAR _Details = {
    -- Receita: positivo = bom → ValorCor igual ao valor
    ("Receita", "vs. Ano Ant.", _VarRec,    "0.0%",  _VarRec),
    -- Custo: positivo = ruim → ValorCor negado para inverter cor
    ("Custo",   "vs. Ano Ant.", _VarCusto, "0.0%",  -_VarCusto)
}

RETURN _HTML.CSS.CARD.CONSTRUCTOR(
    _CSS,
    _HTML.CARD.MULTI_LABEL.DETAIL(_Callout, _Labels, _Details)
)
